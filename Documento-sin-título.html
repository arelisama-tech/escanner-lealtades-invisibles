import {

CheckCircle2,

History,

Download
} from 'lucide-react';
import { clsx } from 'clsx';

return twMerge(clsx(inputs));
}

useState(null);
const ritualRef = useRef(null);

localStorage.getItem('contrato_inconsciente');
if (saved) {
try {

if (parsed.nombre) setNombre(parsed.nombre);
} catch (e) {}
}
}, []);

localStorage.setItem('contrato_inconsciente',
JSON.stringify({ nombre }));
setStep('questions');
};

setAnswers(prev => ({ ...prev, [field]: value }));
};

setStep('scanning');

} else if (answers.dinero === 'escasez') {

riqueza fluya hacia mí con facilidad y gozo.");

donde me abandonen para repetir el ciclo de soledad de las
mujeres/hombres de mi linaje.");

antes de que sea visible, para no atraer el peligro o la envidia que
mis ancestros tanto temían.");

sistema el miedo a brillar. Hoy ocupo mi lugar en el mundo con total
seguridad y confianza en mis talentos.");
contrato.fuerzaBase += 30;

contrato.clausulas.push("Permaneceré pequeño/a
e invisible para protegerme, acatando el mandato de silencio de mi
sistema familiar.");
contrato.decretosSanacion.push("Rompo el voto
de invisibilidad. Me doy permiso de ser visto/a, reconocido/a y de
alzar mi voz con fuerza.");
contrato.fuerzaBase += 30;
}
// Si todo es positivo
if (contrato.clausulas.length === 0) {
contrato.isLibre = true;
contrato.fuerza = 0;
} else {

parezca más orgánico (ej. 60 + 5 = 65%)
contrato.isLibre = false;

9);

}

}, 2500);

${nombre.toUpperCase()}\n\n` +
`"Yo, ${nombre}, en este acto de

`- ${d}`).join('\n') +
`\n\nHoy elijo mi propio destino. Así es y

`Decreto_Liberacion_${nombre.replace(/\s+/g, '_')}.txt`;

setNombre(e.target.value)}
/>
<button
onClick={startScanner}
disabled={!nombre}
className="w-full py-4
bg-indigo-600 text-white rounded-xl font-bold flex items-center
justify-center gap-2 hover:bg-indigo-700 transition-all

>
Iniciar Diagnóstico <ArrowRight
size={20} />
</button>
</div>
</div>
)}
{/* PASO 2: PREGUNTAS (CON OPCIONES POSITIVAS) */}
{step === 'questions' && (
<div className="bg-white rounded-3xl p-6
md:p-10 shadow-xl border border-slate-100 animate-fade-in">
<div className="flex items-center
justify-between mb-8 border-b border-slate-100 pb-4">
<div className="flex items-center
gap-3">
<History
className="text-indigo-500" />
<span className="text-sm font-bold
uppercase tracking-widest text-slate-500">Mapeo de Patrones</span>
</div>
<span className="text-sm text-slate-400
font-medium">Paciente: {nombre}</span>
</div>
<div className="space-y-10">
{/* ÁREA: DINERO */}
<section>
<label className="block text-base

<div className="grid grid-cols-1

<button

handleAnswer('dinero', 'fluidez')}

text-left rounded-xl border transition-all flex items-center gap-3",

border-slate-200 hover:bg-slate-100 text-slate-700")}
>
<div className={cn("w-4 h-4

Fluye con relativa
facilidad y hay estabilidad
</button>

onClick={() =>
handleAnswer('dinero', 'escasez')}

text-left rounded-xl border transition-all flex items-center gap-3",
answers.dinero === 'escasez' ? "bg-indigo-50 border-indigo-500
text-indigo-800 ring-1 ring-indigo-500" : "bg-slate-50 border-slate-200
hover:bg-slate-100 text-slate-700")}
>
<div className={cn("w-4 h-4
rounded-full border-2", answers.dinero === 'escasez' ?

bg-white")}></div>
Siempre falta o hay
constante preocupación y deudas
</button>

onClick={() =>
handleAnswer('dinero', 'sacrificio')}

text-left rounded-xl border transition-all flex items-center gap-3",

hover:bg-slate-100 text-slate-700")}
>
<div className={cn("w-4 h-4
rounded-full border-2", answers.dinero === 'sacrificio' ?
"border-indigo-500 bg-indigo-500" : "border-slate-300
bg-white")}></div>
Llega el dinero, pero solo

</button>
</div>
</section>

<label className="block text-base
font-bold text-slate-800 mb-4">2. En tus relaciones de pareja (o
intentos de tenerla), sueles experimentar...</label>
<div className="grid grid-cols-1
gap-3">
<button
onClick={() =>
handleAnswer('amor', 'sano')}
className={cn("p-4
text-left rounded-xl border transition-all flex items-center gap-3",
answers.amor === 'sano' ? "bg-emerald-50 border-emerald-500
text-emerald-800 ring-1 ring-emerald-500" : "bg-slate-50
border-slate-200 hover:bg-slate-100 text-slate-700")}
>
<div className={cn("w-4 h-4

handleAnswer('amor', 'abandono')}
className={cn("p-4

answers.amor === 'abandono' ? "bg-indigo-50 border-indigo-500

>
<div className={cn("w-4 h-4

</button>
<button

handleAnswer('amor', 'conflicto')}

text-left rounded-xl border transition-all flex items-center gap-3",
answers.amor === 'conflicto' ? "bg-indigo-50 border-indigo-500

>
<div className={cn("w-4 h-4
rounded-full border-2", answers.amor === 'conflicto' ?
"border-indigo-500 bg-indigo-500" : "border-slate-300
bg-white")}></div>
Mucho drama, discusiones
constantes o patrones tóxicos repetitivos
</button>
</div>
</section>

<label className="block text-base

<div className="grid grid-cols-1
gap-3">
<button
onClick={() =>
handleAnswer('exito', 'confianza')}
className={cn("p-4
text-left rounded-xl border transition-all flex items-center gap-3",
answers.exito === 'confianza' ? "bg-emerald-50 border-emerald-500
text-emerald-800 ring-1 ring-emerald-500" : "bg-slate-50
border-slate-200 hover:bg-slate-100 text-slate-700")}
>
<div className={cn("w-4 h-4
rounded-full border-2", answers.exito === 'confianza' ?
"border-emerald-500 bg-emerald-500" : "border-slate-300
bg-white")}></div>
Avanzo con seguridad y
permito que los resultados lleguen
</button>
<button
onClick={() =>
handleAnswer('exito', 'miedo')}
className={cn("p-4
text-left rounded-xl border transition-all flex items-center gap-3",
answers.exito === 'miedo' ? "bg-indigo-50 border-indigo-500
text-indigo-800 ring-1 ring-indigo-500" : "bg-slate-50 border-slate-200
hover:bg-slate-100 text-slate-700")}
>
<div className={cn("w-4 h-4
rounded-full border-2", answers.exito === 'miedo' ? "border-indigo-500
bg-indigo-500" : "border-slate-300 bg-white")}></div>
Me entra un miedo
paralizante, ansiedad o me auto-saboteo
</button>
<button
onClick={() =>
handleAnswer('exito', 'invisibilidad')}
className={cn("p-4
text-left rounded-xl border transition-all flex items-center gap-3",
answers.exito === 'invisibilidad' ? "bg-indigo-50 border-indigo-500
text-indigo-800 ring-1 ring-indigo-500" : "bg-slate-50 border-slate-200
hover:bg-slate-100 text-slate-700")}
>
<div className={cn("w-4 h-4
rounded-full border-2", answers.exito === 'invisibilidad' ?
"border-indigo-500 bg-indigo-500" : "border-slate-300
bg-white")}></div>
Siento que es mejor no
destacar mucho para estar "a salvo"
</button>
</div>
</section>
<button
onClick={generateContract}
disabled={!answers.dinero ||
!answers.amor || !answers.exito}
className="w-full py-4 bg-slate-900
text-white rounded-xl font-bold text-lg hover:bg-indigo-600
transition-all disabled:opacity-50 disabled:cursor-not-allowed
shadow-xl flex items-center justify-center gap-2"
>
Escanear mi Árbol Genealógico
</button>
</div>
</div>
)}
{/* PASO 3: SCANNING */}
{step === 'scanning' && (
<div className="bg-white rounded-3xl p-10
shadow-2xl border border-slate-100 animate-fade-in text-center flex
flex-col items-center justify-center min-h-[400px]">
<div className="relative w-28 h-28 mx-auto

<div className="absolute inset-0

<div className="absolute inset-0

<div className="absolute inset-0

'reverse' }}></div>
<div className="absolute inset-0 flex

<ShieldAlert

</div>
</div>
<h2 className="text-2xl font-bold mb-3

<p className="text-slate-500 animate-pulse
text-lg">Cruzando tus respuestas con el campo sistémico familiar...</p>
</div>

{/* PASO 4: RESULTADO DEL CONTRATO */}
{step === 'result' && scannedContract && (
<div className="animate-fade-in">
{scannedContract.isLibre ? (
// CASO: NO HAY CONTRATOS (Todo
Positivo)
<div className="bg-white rounded-3xl
p-10 shadow-xl border-t-8 border-emerald-500 text-center">
<div className="w-20 h-20
bg-emerald-100 rounded-full flex items-center justify-center mx-auto
mb-6">
<HeartHandshake
className="text-emerald-600 w-10 h-10" />
</div>
<h2 className="text-2xl font-bold

<p className="text-slate-600
text-lg mb-8">
Según tus respuestas, no hemos
detectado contratos inconscientes mayores de carencia, abandono o

</p>
<button
onClick={() =>
window.location.reload()}
className="py-3 px-8
bg-slate-100 text-slate-700 rounded-xl font-bold hover:bg-slate-200

>
Volver al inicio
</button>
</div>
) : (
// CASO: HAY CONTRATOS DETECTADOS
<div className="bg-white rounded-3xl
shadow-2xl border-t-8 border-indigo-900 overflow-hidden contract-bg">
<div className="p-8 text-center
border-b border-slate-200 bg-white/80 backdrop-blur-sm">
<div className="bg-red-100

<AlertCircle size={16} />
ALERTA DE LEALTAD DETECTADA
</div>
<h2 className="text-3xl

<p className="text-sm
text-slate-500 mt-2 uppercase tracking-widest font-semibold">Emitido
para: {nombre}</p>
</div>
<div className="p-6 md:p-10
space-y-8 bg-white/60">
<p className="text-base
text-slate-700 leading-relaxed">
<strong>{nombre}</strong>,
por lealtad ciega a tu sistema familiar, has aceptado inconscientemente

</p>
<ul className="space-y-4">
{scannedContract.clausulas.map((c, i) => (
<li key={i}

<div

<XCircle
className="text-red-500 mt-0.5 shrink-0" size={22} />
<span

<div className="relative

<div className="w-full

<div

style={{ width:
`${scannedContract.fuerza}%` }}
></div>
</div>
<p className="text-sm
mt-6 text-slate-300 leading-relaxed max-w-sm mx-auto">
Esta es la cantidad

{step === 'ritual' && scannedContract && (

<div className="p-6 md:p-10 space-y-8">
<div className="bg-amber-50 border

Instrucciones de Sanación
</h3>
<p className="text-amber-800

Busca un lugar tranquilo.
Respira profundo tres veces. Lee el siguiente decreto <strong>en voz

</p>
</div>
<div className="bg-slate-50 border

<div className="absolute -top-3
