                        const opt = q.options.find(o => o.value === answers[q.id]);

                            clauses.push(opt.clause);

                            decrees.push(opt.decree);

                            weight += 30;

                        }

                    });

                    setContract({

                        isLibre: clauses.length === 0,

                        clauses,

                        decrees,

                    });

            };

                setFinalDecree(intro + body + footer);

                                    </button>

                                </div>
